<!DOCTYPE html>

<html>

<head>

  <title>YouTube Playlist Embed</title>

</head>

<body>

  <div style="position: relative; padding-bottom: 56.25%;">

    <iframe

      id="player"

      style="position: absolute; top: 0; left: 0; width: 100%; height: 100%;"

      frameborder="0"

      allow="autoplay; encrypted-media"

      allowfullscreen

    ></iframe>

  </div>


  <div>

    <label for="playlistId">Playlist ID:</label>

    <input type="text" id="playlistId" placeholder="Enter playlist ID">

    <button onclick="setPlaylistSrc()">Embed</button>

  </div>


  <script>

    // iframeのsrc属性をプレイリストのURLに動的に変更する関数

    function setPlaylistSrc() {

      var playlistId = document.getElementById("playlistId").value;

      var player = document.getElementById("player");

      player.src = "https://www.youtube.com/embed/videoseries?list=" + playlistId;

    }

  </script>

</body>

</html>
